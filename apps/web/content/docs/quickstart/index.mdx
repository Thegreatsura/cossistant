<Steps>
  <Step>
  ## Install the packages
  Let's start by adding Cossistant to your NextJS project:

```bash
npm install @cossistant/next
```

  </Step>
  <Step>
  ## Add your public API key

Add your public API key to the `NEXT_PUBLIC_COSSISTANT_API_KEY` environment variable.

```bash
NEXT_PUBLIC_COSSISTANT_API_KEY=pk_test_xxxx
```

<Alert variant="info" className="mt-10">

  <AlertTitle>Where to find your public API key?</AlertTitle>
  <AlertDescription>
    You can find your public API key at the end of your onboarding or can access
    it anytime from the settings page.
  </AlertDescription>
</Alert>

  </Step>

  <Step>
  ## Add Cossistant provider

Add the Cossistant `SupportProvider` to the root layout of your app:

```tsx {1,13} showLineNumbers title="app/layout.tsx"
import { SupportProvider } from "@cossistant/next";

import "./globals.css";

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html lang="en">
      <body>
        <SupportProvider>{children}</SupportProvider>
      </body>
    </html>
  );
}
```

<Alert variant="info" className="mt-6">
  <Icon name="help" className="size-4" />
  <AlertTitle>Troubleshooting startup animation resets</AlertTitle>
  <AlertDescription>
    Older SDK builds could remount descendants during widget initialization. If
    you still see startup animations replaying, scope `SupportProvider` to the
    smallest subtree that needs support hooks until you upgrade to a build with
    the remount fix.
  </AlertDescription>
</Alert>

  </Step>

  <Step>
  ## Import support widget styling

Cossistant support widget provides two CSS options depending on your setup:

<Tabs defaultValue="tailwindv4" className="mt-10">
  <TabsList>
    <TabsTrigger value="tailwindv4">Tailwind v4</TabsTrigger>
    <TabsTrigger value="css">Plain CSS</TabsTrigger>
  </TabsList>
  <TabsContent value="tailwindv4">
    If you're using Tailwind CSS v4, import the source file in your CSS to enable full theme customization:

    ````css {3} showLineNumbers title="app/globals.css"

@import "tailwindcss";

@import "@cossistant/next/support.css";

````

This allows you to override variables like `--co-theme-primary` in your own CSS for theming.

<Alert variant="info" className="mt-4">
  <Icon name="help" className="size-4" />
  <AlertTitle>Tailwind v3 users</AlertTitle>
  <AlertDescription>
    Tailwind v3 is not supported. Please use the plain CSS import instead (see "Plain CSS" tab).
  </AlertDescription>
</Alert>

  </TabsContent>
  <TabsContent value="css">
    Import the pre-compiled CSS in your layout - works with any setup, no Tailwind required:

    ```tsx {4} showLineNumbers title="app/layout.tsx"
import { SupportProvider } from "@cossistant/next";

import "./globals.css";
import "@cossistant/next/styles.css";

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <html lang="en">
      <body>
        <SupportProvider>{children}</SupportProvider>
      </body>
    </html>
  );
}
```
  </TabsContent>
</Tabs>

  </Step>

    <Step>
  ## Use support widget

Add the Cossistant `<Support />` widget to your app:

```tsx {1,8} showLineNumbers title="app/page.tsx"
import { Support } from "@cossistant/next";

export default function Page() {
  return (
    <div>
      <h1>You're ready to go!</h1>
      {/* Support widget can be placed anywhere in your app */}
      <Support />
    </div>
  );
}
```

  </Step>
